function doGet(e) {
  // Open the Google Sheet by its name.
  // For a more robust solution, especially if you have multiple files with the same name,
  // consider opening by ID: SpreadsheetApp.openById('YOUR_SPREADSHEET_ID');
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("Sheet1");

  if (!sheet) {
    // Return an error message if the sheet is not found
    return ContentService.createTextOutput("Sheet 'Sheet1' not found.");
  }

  // Get all the values in column E (the 5th column)
  const range = sheet.getRange("E:E");
  const values = range.getValues();

  // Find the last non-empty value by iterating backwards
  let latestValue = ""; // Default to an empty string if no value is found
  for (let i = values.length - 1; i >= 0; i--) {
    if (values[i][0] !== "" && values[i][0] !== null && values[i][0] !== undefined) {
      latestValue = values[i][0];
      break; // Exit the loop once the last value is found
    }
  }

  // Return the single value as text
  return ContentService.createTextOutput(latestValue);
}
